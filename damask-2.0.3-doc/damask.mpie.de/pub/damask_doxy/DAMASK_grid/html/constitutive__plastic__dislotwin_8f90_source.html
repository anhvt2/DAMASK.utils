705  endif
706  endif significantslipstress
707  enddo slipstate

710  - dot_rho_dip_formation &
711  - 2.0_preal*rho_dip_distance_min/prm%b_sl * stt%rho_mob(:,of)*abs(dot_gamma_sl)
